import os import unicodedata import pandas as pd import streamlit as st import streamlit.components.v1 as components # ============================================================ # CONFIGURAÇÕES GERAIS # ============================================================ st.set_page_config( page_title="Mapa Estratégico de Campanha", page_icon="🗺️", layout="wide" ) ARQUIVO_DADOS = "dados_campanha.csv" COLUNAS = [ "municipio", "regiao", "responsavel", "estimativa_votos", "liderancas", "escolas", "instituicoes", "prioridade", "status", "observacoes" ] REGIOES_IBGE = [ "Noroeste Paranaense", "Norte Central Paranaense", "Norte Pioneiro Paranaense", "Centro Ocidental Paranaense", "Centro Oriental Paranaense", "Oeste Paranaense", "Sudoeste Paranaense", "Centro-Sul Paranaense", "Sudeste Paranaense", "Metropolitana de Curitiba" ] PRIORIDADES = [ "Alta", "Média", "Baixa" ] STATUS_OPCOES = [ "Não iniciado", "Em articulação", "Em andamento", "Consolidado", "Atenção" ] # ============================================================ # FUNÇÕES AUXILIARES # ============================================================ def normalizar_texto(texto): """ Remove acentos, espaços duplicados e coloca em minúsculo. Isso ajuda a comparar nomes de municípios sem erro. """ if pd.isna(texto): return "" texto = str(texto) texto = unicodedata.normalize("NFD", texto) texto = texto.encode("ascii", "ignore").decode("utf-8") texto = texto.replace("-", " ") texto = " ".join(texto.split()) return texto.lower().strip() def criar_arquivo_se_nao_existir(): """ Cria o CSV vazio caso ele ainda não exista. """ if not os.path.exists(ARQUIVO_DADOS): df = pd.DataFrame(columns=COLUNAS) df.to_csv(ARQUIVO_DADOS, index=False, encoding="utf-8") def carregar_dados(): """ Carrega os dados do CSV e garante que todas as colunas existam. """ criar_arquivo_se_nao_existir() df = pd.read_csv(ARQUIVO_DADOS, dtype=str, encoding="utf-8").fillna("") for coluna in COLUNAS: if coluna not in df.columns: df[coluna] = "" df = df[COLUNAS] df["estimativa_votos"] = pd.to_numeric( df["estimativa_votos"], errors="coerce" ).fillna(0).astype(int) return df def salvar_dados(df): """ Salva os dados no CSV. """ df = df.copy() for coluna in COLUNAS: if coluna not in df.columns: df[coluna] = "" df = df[COLUNAS] df["estimativa_votos"] = pd.to_numeric( df["estimativa_votos"], errors="coerce" ).fillna(0).astype(int) df.to_csv(ARQUIVO_DADOS, index=False, encoding="utf-8") def upsert_municipio(df, registro): """ Se o município já existir, atualiza. Se não existir, cadastra uma nova linha. """ df = df.copy() municipio_normalizado = normalizar_texto(registro["municipio"]) mascara = df["municipio"].apply(normalizar_texto) == municipio_normalizado if mascara.any(): indice = df.index[mascara][0] for coluna in COLUNAS: df.at[indice, coluna] = registro[coluna] else: novo = pd.DataFrame([registro]) df = pd.concat([df, novo], ignore_index=True) return df def aplicar_filtros(df, regioes, responsaveis, prioridades, status, busca): """ Aplica filtros escolhidos pelo usuário. """ filtrado = df.copy() if regioes: filtrado = filtrado[filtrado["regiao"].isin(regioes)] if responsaveis: filtrado = filtrado[filtrado["responsavel"].isin(responsaveis)] if prioridades: filtrado = filtrado[filtrado["prioridade"].isin(prioridades)] if status: filtrado = filtrado[filtrado["status"].isin(status)] if busca.strip(): busca_normalizada = normalizar_texto(busca) filtrado = filtrado[ filtrado.apply( lambda linha: busca_normalizada in normalizar_texto(linha["municipio"]) or busca_normalizada in normalizar_texto(linha["responsavel"]) or busca_normalizada in normalizar_texto(linha["liderancas"]), axis=1 ) ] return filtrado # ============================================================ # CARREGAMENTO INICIAL # ============================================================ dados = carregar_dados() # ============================================================ # SIDEBAR — CADASTRO # ============================================================ st.sidebar.title("🧭 Painel de Cadastro") st.sidebar.caption( "Cadastre ou atualize as informações estratégicas de cada município." ) with st.sidebar.form("form_cadastro_municipio", clear_on_submit=False): municipio = st.text_input("Município") regiao = st.selectbox( "Região IBGE", REGIOES_IBGE ) responsavel = st.text_input("Responsável pela área") estimativa_votos = st.number_input( "Estimativa de votos", min_value=0, step=100 ) prioridade = st.selectbox( "Prioridade", PRIORIDADES ) status = st.selectbox( "Status da articulação", STATUS_OPCOES ) liderancas = st.text_area( "Lideranças", placeholder="Exemplo: João Silva; Maria Souza; Associação X" ) escolas = st.text_area( "Escolas parceiras", placeholder="Exemplo: Escola Municipal A; Colégio Estadual B" ) instituicoes = st.text_area( "Instituições parceiras", placeholder="Exemplo: Associação Comercial; Sindicato Rural" ) observacoes = st.text_area( "Observações", placeholder="Anotações internas da equipe" ) botao_salvar = st.form_submit_button("💾 Salvar ou atualizar município") if botao_salvar: if not municipio.strip(): st.sidebar.error("Informe o nome do município.") else: registro = { "municipio": municipio.strip(), "regiao": regiao, "responsavel": responsavel.strip(), "estimativa_votos": int(estimativa_votos), "liderancas": liderancas.strip(), "escolas": escolas.strip(), "instituicoes": instituicoes.strip(), "prioridade": prioridade, "status": status, "observacoes": observacoes.strip() } dados = upsert_municipio(dados, registro) salvar_dados(dados) st.sidebar.success("Município salvo com sucesso.") # ============================================================ # SIDEBAR — EXCLUSÃO # ============================================================ st.sidebar.divider() with st.sidebar.expander("🗑️ Excluir cadastro"): municipios_cadastrados = sorted( [m for m in dados["municipio"].dropna().unique() if str(m).strip()] ) if municipios_cadastrados: municipio_excluir = st.selectbox( "Município para excluir", municipios_cadastrados ) confirmar_exclusao = st.checkbox( "Confirmo que desejo excluir este cadastro" ) if st.button("Excluir município"): if confirmar_exclusao: dados = dados[ dados["municipio"].apply(normalizar_texto) != normalizar_texto(municipio_excluir) ] salvar_dados(dados) st.warning("Cadastro excluído. Recarregue a página se necessário.") else: st.error("Marque a confirmação antes de excluir.") else: st.info("Nenhum município cadastrado ainda.") # ============================================================ # SIDEBAR — FILTROS # ============================================================ st.sidebar.divider() st.sidebar.title("🔎 Filtros") filtro_busca = st.sidebar.text_input( "Buscar", placeholder="Município, responsável ou liderança" ) filtro_regioes = st.sidebar.multiselect( "Região", REGIOES_IBGE ) responsaveis_disponiveis = sorted( [r for r in dados["responsavel"].dropna().unique() if str(r).strip()] ) filtro_responsaveis = st.sidebar.multiselect( "Responsável", responsaveis_disponiveis ) filtro_prioridades = st.sidebar.multiselect( "Prioridade", PRIORIDADES ) filtro_status = st.sidebar.multiselect( "Status", STATUS_OPCOES ) dados_filtrados = aplicar_filtros( dados, filtro_regioes, filtro_responsaveis, filtro_prioridades, filtro_status, filtro_busca ) # ============================================================ # CABEÇALHO # ============================================================ st.title("🗺️ Mapa Estratégico de Campanha Política") st.caption( "Sistema territorial para cadastro, visualização e acompanhamento de municípios." ) st.info( "Use dados agregados por município/região. Evite cadastrar dados pessoais de eleitores, alunos ou menores de idade sem autorização e base legal." ) # ============================================================ # INDICADORES # ============================================================ total_municipios = dados_filtrados["municipio"].nunique() total_votos = int(dados_filtrados["estimativa_votos"].sum()) if not dados_filtrados.empty else 0 total_responsaveis = dados_filtrados["responsavel"].replace("", pd.NA).dropna().nunique() total_alta_prioridade = len(dados_filtrados[dados_filtrados["prioridade"] == "Alta"]) col1, col2, col3, col4 = st.columns(4) with col1: st.metric("Municípios cadastrados", total_municipios) with col2: st.metric("Votos estimados", f"{total_votos:,}".replace(",", ".")) with col3: st.metric("Responsáveis ativos", total_responsaveis) with col4: st.metric("Alta prioridade", total_alta_prioridade) # ============================================================ # ABAS PRINCIPAIS # ============================================================ aba_mapa, aba_cadastros = st.tabs([ "🗺️ Mapa Interativo", "📋 Cadastros" ]) # ============================================================ # ABA MAPA # ============================================================ with aba_mapa: dados_json = dados_filtrados.to_json( orient="records", force_ascii=False ) html_mapa = """
Selecione um município
Clique em uma cidade no mapa para visualizar os dados cadastrados.
O mapa mostra todos os municípios do Paraná. As informações aparecem quando houver cadastro salvo no sistema.

Noroeste

Norte Pioneiro
Centro Ocidental
Centro Oriental

Centro-Sul
Sudeste
Metropolitana
""" html_mapa = html_mapa.replace("__DADOS_CAMPANHA__", dados_json) components.html( html_mapa, height=820, scrolling=False ) # ============================================================ # ABA CADASTROS # ============================================================ with aba_cadastros: st.subheader("📋 Cadastros salvos") st.caption( "Você pode editar diretamente na tabela abaixo. Depois, clique em salvar alterações." ) if dados.empty: st.info("Nenhum município cadastrado ainda.") else: dados_editados = st.data_editor( dados, use_container_width=True, num_rows="dynamic", column_config={ "municipio": st.column_config.TextColumn( "Município", required=True ), "regiao": st.column_config.SelectboxColumn( "Região", options=REGIOES_IBGE ), "responsavel": st.column_config.TextColumn( "Responsável" ), "estimativa_votos": st.column_config.NumberColumn( "Estimativa de votos", min_value=0, step=100 ), "liderancas": st.column_config.TextColumn( "Lideranças" ), "escolas": st.column_config.TextColumn( "Escolas parceiras" ), "instituicoes": st.column_config.TextColumn( "Instituições parceiras" ), "prioridade": st.column_config.SelectboxColumn( "Prioridade", options=PRIORIDADES ), "status": st.column_config.SelectboxColumn( "Status", options=STATUS_OPCOES ), "observacoes": st.column_config.TextColumn( "Observações" ) } ) col_salvar, col_download = st.columns([1, 1]) with col_salvar: if st.button("💾 Salvar alterações da tabela"): salvar_dados(dados_editados) st.success("Alterações salvas com sucesso.") with col_download: csv_exportado = dados.to_csv(index=False).encode("utf-8") st.download_button( label="⬇️ Baixar CSV", data=csv_exportado, file_name="dados_campanha.csv", mime="text/csv" ) st.divider() st.subheader("Resumo dos dados filtrados") if dados_filtrados.empty: st.warning("Nenhum cadastro encontrado com os filtros atuais.") else: st.dataframe( dados_filtrados, use_container_width=True )
